#!/usr/bin/env python3
"""
Экспорт групп хостов из Zabbix 6 в JSON-файл.
"""

import json
import sys
import requests

# ===== КОНФИГУРАЦИЯ  =====
ZABBIX_URL = "http://zabbix6.example.com/zabbix/api_jsonrpc.php"
ZABBIX_USER = "Admin"
ZABBIX_PASSWORD = "your_password"
OUTPUT_FILE = "hostgroups_export.json"
VERIFY_SSL = False   # для HTTP можно False
# ============================================

def zabbix_login(url, user, password):
    payload = {
        "jsonrpc": "2.0",
        "method": "user.login",
        "params": {"username": user, "password": password},
        "id": 1,
        "auth": None
    }
    resp = requests.post(url, json=payload, verify=VERIFY_SSL, timeout=30)
    resp.raise_for_status()
    data = resp.json()
    if "error" in data:
        raise Exception(f"Login error: {data['error']['data']}")
    return data["result"]

def get_all_hostgroups(url, token):
    payload = {
        "jsonrpc": "2.0",
        "method": "hostgroup.get",
        "params": {"output": "extend"},
        "id": 2,
        "auth": token
    }
    resp = requests.post(url, json=payload, verify=VERIFY_SSL, timeout=30)
    resp.raise_for_status()
    data = resp.json()
    if "error" in data:
        raise Exception(f"API error: {data['error']['data']}")
    return data["result"]

def main():
    print(f"Connecting to Zabbix 6: {ZABBIX_URL}")
    token = zabbix_login(ZABBIX_URL, ZABBIX_USER, ZABBIX_PASSWORD)
    print("Login successful")

    groups = get_all_hostgroups(ZABBIX_URL, token)
    print(f"Found {len(groups)} host groups")

    # Сохраняем в файл
    with open(OUTPUT_FILE, "w", encoding="utf-8") as f:
        json.dump(groups, f, ensure_ascii=False, indent=2)

    print(f"Exported to {OUTPUT_FILE}")

if __name__ == "__main__":
    try:
        main()
    except Exception as e:
        print(f"ERROR: {e}", file=sys.stderr)
        sys.exit(1)
