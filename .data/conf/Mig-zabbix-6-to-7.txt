#!/usr/bin/env python3
"""
Импорт групп хостов из JSON-файла в Zabbix 7.
Аутентификация через API-токен (Bearer). Поле 'auth' в теле запроса не используется.
"""

import json
import re
import sys
import requests
from requests.packages.urllib3.exceptions import InsecureRequestWarning

requests.packages.urllib3.disable_warnings(InsecureRequestWarning)

# ========================= КОНФИГУРАЦИЯ =========================
ZABBIX_URL = "http://localhost/api_jsonrpc.php"
API_TOKEN = "ваш_токен_здесь"          # Токен, созданный в Zabbix 7

INPUT_FILE = "hostgroups_export.json"
DRY_RUN = False                        # True — только просмотр
VERIFY_SSL = False

# Правила переименования
RENAME_RULES = [
    {"type": "replace", "old": "/Service/", "new": "/Systems/"},
]
# ================================================================


def zabbix_request(url, token, method, params, request_id=1):
    """Универсальная функция для вызова API Zabbix с Bearer-токеном."""
    headers = {"Authorization": f"Bearer {token}"}
    payload = {
        "jsonrpc": "2.0",
        "method": method,
        "params": params,
        "id": request_id,
        # Поле "auth" отсутствует — при токене оно не нужно
    }
    resp = requests.post(url, json=payload, headers=headers, verify=VERIFY_SSL, timeout=30)
    resp.raise_for_status()
    data = resp.json()
    if "error" in data:
        raise Exception(f"API error in {method}: {data['error']['data']}")
    return data["result"]


def get_existing_group_names(url, token):
    """Возвращает множество имён существующих групп хостов."""
    groups = zabbix_request(url, token, "hostgroup.get", {"output": ["name"]}, 2)
    return {g["name"] for g in groups}


def create_hostgroup(url, token, name):
    """Создаёт группу хостов, возвращает True при успехе."""
    try:
        result = zabbix_request(url, token, "hostgroup.create", {"name": name}, 3)
        group_id = result["groupids"][0]
        print(f"  [OK] Created '{name}' (ID: {group_id})")
        return True
    except Exception as e:
        print(f"  [FAIL] Creation failed for '{name}': {e}")
        return False


def apply_rename_rules(original_name, rules):
    new_name = original_name
    for rule in rules:
        if rule["type"] == "replace":
            new_name = new_name.replace(rule["old"], rule["new"])
        elif rule["type"] == "regex":
            new_name = re.sub(rule["pattern"], rule["repl"], new_name)
    return new_name


def main():
    # Загрузка групп из файла
    try:
        with open(INPUT_FILE, "r", encoding="utf-8") as f:
            groups = json.load(f)
    except Exception as e:
        print(f"ERROR reading file: {e}", file=sys.stderr)
        sys.exit(1)

    print(f"Loaded {len(groups)} groups from {INPUT_FILE}")

    # Преобразование имён (уникальные новые имена)
    transformed = {}
    for g in groups:
        old = g["name"]
        new = apply_rename_rules(old, RENAME_RULES)
        transformed[new] = old

    print("\nName transformation preview (first 10):")
    for i, (new, old) in enumerate(transformed.items()):
        if i >= 10:
            print("  ... and more")
            break
        if new != old:
            print(f"  {old} -> {new}")
        else:
            print(f"  {old} (no change)")

    if DRY_RUN:
        print("\n--- DRY RUN: no changes will be made ---")
        print(f"Would create {len(transformed)} groups")
        return

    # Подключение к Zabbix 7 через токен
    print(f"\nConnecting to Zabbix 7 using API token: {ZABBIX_URL}")
    existing = get_existing_group_names(ZABBIX_URL, API_TOKEN)
    print(f"Existing groups in Zabbix 7: {len(existing)}")

    # Создание недостающих групп
    created = 0
    skipped = 0
    failed = 0

    for new_name in transformed.keys():
        if new_name in existing:
            print(f"  [SKIP] Skipping '{new_name}' (already exists)")
            skipped += 1
            continue
        if create_hostgroup(ZABBIX_URL, API_TOKEN, new_name):
            created += 1
            existing.add(new_name)
        else:
            failed += 1

    print("\n" + "=" * 50)
    print("IMPORT SUMMARY")
    print(f"  Total unique transformed names: {len(transformed)}")
    print(f"  [OK] Created: {created}")
    print(f"  [SKIP] Skipped (already exist): {skipped}")
    print(f"  [FAIL] Failed: {failed}")
    print("=" * 50)


if __name__ == "__main__":
    try:
        main()
    except Exception as e:
        print(f"ERROR: {e}", file=sys.stderr)
        sys.exit(1)
